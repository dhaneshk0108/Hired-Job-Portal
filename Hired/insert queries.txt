Company and contact details
----------------------------
SELECT COMPANYDETAILS.COMPANYNAME,
  PERSON.FIRSTNAME,
  PERSON.LASTNAME,
  PERSON.EMAILID,
  JOBDETAILS.JOBTITLE,
  JOBDETAILS.JOBPOSTEDDATE,
  JOBSTATUS.STATUSTYPE
FROM COMPANYDETAILS
INNER JOIN EMPLOYERDETAILS
ON COMPANYDETAILS.ID = EMPLOYERDETAILS.COMPANYID
INNER JOIN PERSON
ON PERSON.ID = EMPLOYERDETAILS.PERSONID
INNER JOIN JOBDETAILS
ON JOBDETAILS.ID = EMPLOYERDETAILS.JOBID
INNER JOIN JOBSTATUS
ON JOBSTATUS.ID = JOBDETAILS.STATUSID
